create a new Playlist: create(object)
find a Playlist by id: findByPk(id)
get all Playlist: findAll()
update a Playlist by id: update(data, where: { id: id })
remove a Playlist: destroy(where: { id: id })

find all Playlists by title: findAll({ where: { title: ... } })
